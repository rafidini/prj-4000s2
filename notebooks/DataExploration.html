# Function that read and display an image
def read_display(index):
    img   = cv.imread(filename=f"data/images/{train.loc[index, 'image_file_name']}")
    label = train.loc[index, 'color_tags']

    if img is not None:
        colored = cv.cvtColor(img, cv.COLOR_BGR2RGB)
        plt.imshow(colored)
        plt.title(str(label))
        plt.show()

# Example
for i in range(3):
    read_display(i)

Les images d'articles comportent des fonds colors interfrant avec la prdiction de la couleurs des objets.
